import java.util.*;

// Interfaces
interface Fly {
    void fly();
}

interface Eat {
    void eat();
}

interface SaySomething {
    void saySomething();
}

interface FlyingBehaviour {
    void flyingBehaviour();
}

// FlyingBehaviour Implementations
class PigeonSparrowFlyingBehaviour implements FlyingBehaviour {
    @Override
    public void flyingBehaviour() {
        System.out.println("Flying low and quick.");
    }
}

class CrowEagleFlyingBehaviour implements FlyingBehaviour {
    @Override
    public void flyingBehaviour() {
        System.out.println("Flying high and strong.");
    }
}

// Base Bird Class
class Bird {
    private int weight;
    private String name;
    private String color;

    public void setColor(String color) {
        this.color = color;
    }

    public void setWeight(int weight) {
        this.weight = weight;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getColor() {
        return color;
    }

    public String getName() {
        return name;
    }

    public int getWeight() {
        return weight;
    }
}

// Bird Subclasses
class Sparrow extends Bird implements Fly, Eat {
    @Override
    public void eat() {
        System.out.println("Sparrow eats grains.");
    }

    @Override
    public void fly() {
        System.out.println("Sparrow is flying.");
    }
}

class Crow extends Bird implements Fly, Eat, SaySomething {
    @Override
    public void eat() {
        System.out.println("Crow eats leftovers.");
    }

    @Override
    public void fly() {
        System.out.println("Crow is flying.");
    }

    @Override
    public void saySomething() {
        System.out.println("Crow says Caw Caw!");
    }

    @Override
    public String getColor() {
        return super.getColor() + " (Crow specific)";
    }
}

class Pigeon extends Bird implements Fly {
    FlyingBehaviour flyingBehaviour;

    public Pigeon() {
        this.flyingBehaviour = new PigeonSparrowFlyingBehaviour();
    }

    @Override
    public void fly() {
        flyingBehaviour.flyingBehaviour();
    }
}

class Eagle extends Bird implements Fly, Eat, SaySomething {
    @Override
    public void eat() {
        System.out.println("Eagle eats meat.");
    }

    @Override
    public void fly() {
        System.out.println("Eagle is soaring.");
    }

    @Override
    public void saySomething() {
        System.out.println("Eagle screams!");
    }
}

class Dolphin implements Eat, SaySomething {
    @Override
    public void eat() {
        System.out.println("Dolphin eats fish.");
    }

    @Override
    public void saySomething() {
        System.out.println("Dolphin clicks and whistles.");
    }
}

// Main Class
public class Main {
    public static void main(String[] args) {
        List<Fly> allFlyingBirds = new ArrayList<>();

        Fly sparrow = new Sparrow();
        Fly crow = new Crow();
        Fly pigeon = new Pigeon();
        Fly eagle = new Eagle();

        allFlyingBirds.add(sparrow);
        allFlyingBirds.add(crow);
        allFlyingBirds.add(pigeon);
        allFlyingBirds.add(eagle);

        System.out.println("Flying Birds:");
        for (Fly bird : allFlyingBirds) {
            bird.fly();
        }

        // Example for Dolphin and Crow
        Dolphin dolphin = new Dolphin();
        Crow exampleCrow = new Crow();

        System.out.println("\nOther actions:");
        dolphin.eat();
        dolphin.saySomething();
        exampleCrow.saySomething();
        exampleCrow.eat();
    }
}
